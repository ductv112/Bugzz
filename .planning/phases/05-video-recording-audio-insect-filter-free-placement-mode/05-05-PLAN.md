**Layer Box (9 layers per UI-SPEC §5):**
```kotlin
@Composable
fun InsectFilterScreen(
    viewModel: InsectFilterViewModel = hiltViewModel(),
    navController: NavController,
) {
    // CAMERA permission gate (reuse Phase 1 D-13 pattern from CameraScreen)
    // ... (omitted; identical block)

    val uiState by viewModel.uiState.collectAsStateWithLifecycle()
    val surfaceRequest by viewModel.surfaceRequest.collectAsStateWithLifecycle()
    val isRecording = uiState.recordingState is RecordingState.Active
    val isStopping = uiState.recordingState is RecordingState.Stopping
    val elapsedMs = (uiState.recordingState as? RecordingState.Active)?.elapsedMs ?: 0L

    val view = LocalView.current
    val context = LocalContext.current

    var showDiscardDialog by rememberSaveable { mutableStateOf(false) }

    // RECORD_AUDIO launcher (identical to CameraScreen Plan 04)
    val audioPermissionLauncher = rememberLauncherForActivityResult(
        contract = ActivityResultContracts.RequestPermission()
    ) { granted ->
        if (granted) viewModel.onRecordTapped(audioEnabled = true)
        // else show snackbar (identical Plan 04 pattern)
    }

    BackHandler(enabled = isRecording) { showDiscardDialog = true }

    LaunchedEffect(Unit) {
        viewModel.events.collect { event ->
            when (event) {
                is OneShotEvent.PhotoSaved -> Toast.makeText(context, "Saved to gallery", Toast.LENGTH_SHORT).show()
                is OneShotEvent.PhotoError -> Toast.makeText(context, "Photo failed: ${event.reason}", Toast.LENGTH_SHORT).show()
                is OneShotEvent.VideoSaved -> Toast.makeText(context, "Recording saved", Toast.LENGTH_SHORT).show()
                is OneShotEvent.VideoError -> Toast.makeText(context, "Recording failed: ${event.message}", Toast.LENGTH_LONG).show()
                is OneShotEvent.FilterLoadError -> Toast.makeText(context, "Filter failed to load", Toast.LENGTH_SHORT).show()
                else -> Unit
            }
        }
    }

    Box(modifier = Modifier.fillMaxSize().background(Color.Black)) {
        // LAYER 1: CameraXViewfinder (preview)
        surfaceRequest?.let { req ->
            CameraXViewfinder(
                surfaceRequest = req,
                implementationMode = ImplementationMode.EXTERNAL,
                modifier = Modifier
                    .fillMaxSize()
                    .onGloballyPositioned { coords ->
                        val size = coords.size
                        // bitmapSize = active sticker bitmap dims; default to 200x200 placeholder
                        // (StickerRenderer overrides via setActiveFilter — bitmap size known on first frame)
                        viewModel.onPreviewSizeChanged(
                            size = IntSize(size.width, size.height),
                            bitmapSize = IntSize(200, 200),  // placeholder; corrected on render
                        )
                    }
            )
        }

        // LAYER 2: Sticker overlay — drawn by StickerRenderer in OverlayEffect canvas (NOT a Compose layer per UI-SPEC §6)
        // No composable here — sticker comes from OverlayEffect output baked into preview surface

        // LAYER 3: Full-screen gesture handler (above viewfinder, below buttons)
        Box(
            modifier = Modifier
                .matchParentSize()
                .pointerInput(isRecording) {
                    if (isRecording) return@pointerInput  // D-23 lock
                    detectTransformGestures { _, pan, zoom, rotation ->
                        viewModel.onStickerGesture(
                            pan = pan,
                            zoom = zoom,
                            rotationDelta = rotation,
                        )
                    }
                }
                .semantics { invisibleToUser() }
        )

        // LAYER 4: Filter picker LazyRow (reuse Phase 4 FilterPicker)
        FilterPicker(
            filters = uiState.filters ?: emptyList(),  // adapt VM API
            selectedId = uiState.selectedFilterId,
            onFilterSelected = { id ->
                if (!isRecording) viewModel.onFilterSelected(id)
            },
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .padding(bottom = 104.dp)
                .alpha(if (isRecording) 0.5f else 1f),
        )

        // LAYER 5: Shutter button — INLINED (matches CameraScreen pattern; option B per WARNING 9)
        Box(
            modifier = Modifier
                .align(Alignment.BottomCenter)
                .padding(bottom = 24.dp)
                .size(72.dp)
                .clip(CircleShape)
                .background(Color.White)
                .clickable {
                    view.performHapticFeedback(android.view.HapticFeedbackConstants.LONG_PRESS)
                    viewModel.onShutterTapped()
                }
                .semantics {
                    role = Role.Button
                    contentDescription = "Take photo"
                }
        )

        // LAYER 6: Production Record button — REUSE extracted composable from Plan 05-04
        RecordButton(
            isRecording = isRecording,
            isStopping = isStopping,
            onTap = {
                view.performHapticFeedback(android.view.HapticFeedbackConstants.VIRTUAL_KEY)
                if (isRecording) viewModel.onStopRecording()
                else {
                    val granted = ContextCompat.checkSelfPermission(
                        context, android.Manifest.permission.RECORD_AUDIO
                    ) == PackageManager.PERMISSION_GRANTED
                    if (granted) viewModel.onRecordTapped(audioEnabled = true)
                    else audioPermissionLauncher.launch(android.Manifest.permission.RECORD_AUDIO)
                }
            },
            modifier = Modifier
                .align(Alignment.BottomStart)
                .padding(start = 24.dp, bottom = 24.dp),
        )

        // LAYER 7: Flip button (TopEnd) — disabled during record
        OutlinedButton(
            onClick = { viewModel.onFlipLens() },
            enabled = !isRecording,
            modifier = Modifier
                .align(Alignment.TopEnd)
                .padding(top = 16.dp, end = 16.dp),
        ) { Text("Flip") }

        // LAYER 8: Recording indicator — REUSE extracted composable from Plan 05-04
        RecordingIndicator(
            isRecording = isRecording,
            elapsedMs = elapsedMs,
            modifier = Modifier.align(Alignment.TopCenter).padding(top = 24.dp),
        )

        // LAYER 9: Capture flash overlay (Phase 3 D-16 inherited)
        AnimatedVisibility(visible = uiState.captureFlashVisible, ...) { ... }
    }

    // AlertDialog (identical to CameraScreen)
    if (showDiscardDialog) {
        AlertDialog(
            onDismissRequest = { showDiscardDialog = false },
            title = { Text("Recording in progress", style = TextStyle(fontSize = 16.sp, fontWeight = FontWeight.Medium)) },
            text = { Text("Are you sure you want to discard this recording?", style = MaterialTheme.typography.bodyMedium) },
            confirmButton = { TextButton(onClick = { showDiscardDialog = false }) { Text("Cancel") } },
            dismissButton = {
                TextButton(
                    onClick = { showDiscardDialog = false; viewModel.onDiscardRecording() },
                    colors = ButtonDefaults.textButtonColors(contentColor = MaterialTheme.colorScheme.error),
                ) { Text("Discard") }
            }
        )
    }
}
```

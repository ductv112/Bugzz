```javascript
// scripts/extract_sprites.cjs
const fs = require('fs');
const path = require('path');

// 4 output groups per 04-CONTEXT D-02 amended roster. assetIds null = take all assets with non-empty .p
// (used for spider_prankfilter.json which is flat); explicit arrays for home_lottie.json split.
const GROUPS = [
  { file: 'reference/raw_extract/res/raw/spider_prankfilter.json',
    outputDir: 'sprite_spider', assetIds: null },
  { file: 'reference/raw_extract/res/raw/home_lottie.json',
    outputDir: 'sprite_bugA',   assetIds: ['imgSeq_0','imgSeq_1','imgSeq_2','imgSeq_3','imgSeq_4','imgSeq_5','imgSeq_6'] },
  { file: 'reference/raw_extract/res/raw/home_lottie.json',
    outputDir: 'sprite_bugB',   assetIds: ['imgSeq_14','imgSeq_15','imgSeq_16','imgSeq_17','imgSeq_18','imgSeq_19','imgSeq_20','imgSeq_21','imgSeq_22','imgSeq_23','imgSeq_24','imgSeq_25'] },
  { file: 'reference/raw_extract/res/raw/home_lottie.json',
    outputDir: 'sprite_bugC',   assetIds: ['imgSeq_38','imgSeq_39','imgSeq_40','imgSeq_41','imgSeq_42','imgSeq_43','imgSeq_44','imgSeq_45','imgSeq_46','imgSeq_47','imgSeq_48','imgSeq_49','imgSeq_50','imgSeq_51','imgSeq_52','imgSeq_53'] },
];

const BASE = 'app/src/main/assets/sprites';
const MIN_DENSITY = 0.10;  // D-05: non-alpha density heuristic (bytes / (w*h*4))

for (const { file, outputDir, assetIds } of GROUPS) {
  const data = JSON.parse(fs.readFileSync(file, 'utf8'));
  const fr = data.fr || 24;
  const frameDurationMs = Math.round(1000 / fr);

  const chosen = data.assets.filter(a =>
    a.p && a.p.length > 50 &&
    (assetIds === null || assetIds.includes(a.id))
  );

  console.log(`[${outputDir}] matched ${chosen.length} assets from ${path.basename(file)}`);

  const outDir = path.join(BASE, outputDir);
  fs.mkdirSync(outDir, { recursive: true });

  let extracted = 0;
  chosen.forEach((asset, idx) => {
    const b64 = asset.p.replace(/^data:image\/png;base64,/, '');
    const buf = Buffer.from(b64, 'base64');

    // D-05 density heuristic — reject mostly-transparent sprites.
    const pixelCount = (asset.w || 360) * (asset.h || 360);
    const density = buf.length / (pixelCount * 4);
    if (density < MIN_DENSITY) {
      console.warn(`  SKIP ${outputDir}/frame_${String(idx).padStart(2,'0')}.png (id=${asset.id}, density ${density.toFixed(3)} < ${MIN_DENSITY})`);
      return;
    }

    const fname = `frame_${String(idx).padStart(2,'0')}.png`;
    fs.writeFileSync(path.join(outDir, fname), buf);
    extracted++;
    console.log(`  OK   ${outputDir}/${fname} (id=${asset.id}, ${buf.length} bytes, density ${density.toFixed(3)})`);
  });

  // Manifest — D-29 optional behaviorConfig omitted here (Plan 04-04 emits per-filter manifests if needed);
  // these shared-group manifests carry neutral defaults since FilterDefinition overrides them (D-30).
  const manifest = {
    id: outputDir,
    displayName: outputDir.replace(/_/g, ' '),
    frameCount: extracted,
    frameDurationMs,
    anchorType: 'NOSE_TIP',        // placeholder — overridden by FilterDefinition.anchorType per D-30
    behavior: 'STATIC',             // placeholder — overridden by FilterDefinition.behavior per D-30
    scaleFactor: 0.22,
    mirrorable: true
  };
  fs.writeFileSync(path.join(outDir, 'manifest.json'), JSON.stringify(manifest, null, 2));
  console.log(`[${outputDir}] ${extracted}/${chosen.length} frames extracted → manifest.json (${frameDurationMs}ms/frame)\n`);
}
```

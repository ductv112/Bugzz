<!-- SplashViewModel shape (RESEARCH Code Examples §SplashScreen pattern) -->
@HiltViewModel
class SplashViewModel @Inject constructor(
    private val prefs: FilterPrefsRepository,
) : ViewModel() {
    /**
     * StateFlow<Boolean?> — null = loading (DataStore not yet read),
     * false = first launch (route to Onboarding), true = subsequent (route to Home).
     */
    val onboardingCompleted: StateFlow<Boolean?> = prefs.onboardingCompleted
        .map<Boolean, Boolean?> { it }
        .catch { emit(null) }
        .stateIn(viewModelScope, SharingStarted.Eagerly, null)
}

<!-- OnboardingViewModel shape -->
@HiltViewModel
class OnboardingViewModel @Inject constructor(
    private val prefs: FilterPrefsRepository,
) : ViewModel() {
    /**
     * Persists onboarding_completed=true. Caller (composable) navigates to HomeRoute on complete.
     * Suspend → composable wraps in viewModelScope/rememberCoroutineScope OR uses fire-and-forget launch.
     */
    fun completeOnboarding() {
        viewModelScope.launch { prefs.setOnboardingCompleted() }
    }
}
